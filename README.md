@Dao
interface CarDao {

    @Query("SELECT * FROM Car")
    fun getAllCars() : LiveData<List<Car>>

    @Upsert
    fun addCar(car : Car)

    @Query("Delete FROM CAR WHERE id = :id")
    fun deleteCar(id : Int)
}
